### 5.2 变更影响分析规则

```yaml
# 05-cross-reference/impact-analysis/change-impact-rules.yaml

change_impact_rules:
  version: "1.0"

  rules:

    # 规则1：API 变更影响分析
    - rule_id: "CIR-001"
      trigger:
        change_type: "api-breaking-change"
        target_type: "api-endpoint"
      impact_propagation:
        - direction: "upstream"
          description: "影响所有调用此 API 的上游服务"
          lookup: "dependency-graph.edges[type='depends-on' AND to=target]"
          severity: "critical"
        - direction: "cross-view-product"
          description: "影响此 API 支撑的功能点和用例"
          lookup: "product-to-technical.feature_to_api[apis.api_id=target]"
          severity: "high"
        - direction: "cross-view-business"
          description: "影响关联的业务流程"
          lookup: "product-to-technical.usecase_to_service_chain → business-to-product.domain_event_to_process_step"
          severity: "medium"
      required_actions:
        - "通知所有上游服务团队"
        - "评估 API 版本兼容性策略"
        - "更新 API changelog"
        - "更新产品→技术映射矩阵"

    # 规则2：数据库 Schema 变更影响分析
    - rule_id: "CIR-002"
      trigger:
        change_type: "schema-change"
        target_type: "data-entity"
      impact_propagation:
        - direction: "cross-view-technical"
          description: "影响所有访问此表的微服务"
          lookup: "technical-to-data.service_to_datastore[entities_accessed contains target]"
          severity: "critical"
        - direction: "cross-view-business"
          description: "影响对应的聚合和领域模型"
          lookup: "technical-to-data.aggregate_to_physical_table[physical_tables.table=target]"
          severity: "high"
        - direction: "data-downstream"
          description: "影响下游数据流（CDC、ETL）"
          lookup: "technical-to-data.event_to_dataflow[data_entities contains target]"
          severity: "high"
        - direction: "cross-view-product"
          description: "影响读写此数据的功能点"
          lookup: "product-to-data.feature_to_data_access[data_reads/data_writes.entity=target]"
          severity: "medium"
      required_actions:
        - "执行数据库迁移兼容性检查"
        - "通知所有数据消费方"
        - "更新数据字典"
        - "更新聚合→物理表映射"
        - "验证数据流转链路"

    # 规则3：领域事件变更影响分析

      trigger:
        change_type: "event-schema-change"
        target_type: "domain-event"

        - direction: "technical-subscribers"
          description: "影响所有订阅此事件的微服务"
          lookup: "dependency-graph.edges[type='subscribes' AND from=target]"
          severity: "critical"
        - direction: "cross-view-product"
          description: "影响此事件触发的业务流程步骤"
          lookup: "business-to-product.domain_event_to_process_step[domain_event=target]"
          severity: "high"
        - direction: "data-propagation"
          description: "影响此事件驱动的数据流"
          lookup: "technical-to-data.event_to_dataflow[event=target]"
          severity: "high"
        - direction: "cross-view-data"
          description: "影响此事件引发的数据变更"
          lookup: "business-to-data.domain_event_to_data_mutation[domain_event=target]"
          severity: "medium"
      required_actions:
        - "通知所有事件消费者团队"
        - "评估事件版本兼容性策略"
        - "更新 AsyncAPI 规约"
        - "验证数据流转完整性"

        target_type: "business-rule"
```

### 5.3 爆炸半径模板

```yaml
# 05-cross-reference/impact-analysis/blast-radius-templates/api-change-blast.yaml

blast_radius_template:
  template_id: "BRT-API-CHANGE"
  name: "API 变更爆炸半径分析模板"
  version: "1.0"

  input:
    required:
      - name: "api_id"
        description: "变更的 API 端点 ID"
      - name: "change_type"
        description: "变更类型"
        enum: ["field-added", "field-removed", "field-type-changed", "endpoint-removed", "endpoint-path-changed", "response-structure-changed"]
      - name: "is_breaking"
        description: "是否为破坏性变更"
        type: "boolean"

  analysis_steps:

    - step: 1
      name: "识别直接影响的服务"
      action: "查询依赖关系图，找到所有调用此 API 的上游服务"
      query: |
        SELECT upstream_service 
        FROM dependency_graph 
        WHERE downstream_api = {api_id}
      output: "affected_services"

    - step: 2
      name: "识别影响的功能点"
      action: "通过产品→技术映射，找到此 API 支撑的功能点"
      query: |
        SELECT feature_id, involvement 
        FROM product_to_technical.feature_to_api 
        WHERE api_id = {api_id}
      output: "affected_features"

    - step: 3
      name: "识别影响的用例和业务流程"
      action: "通过功能点→用例→业务流程链路追溯"
      query: |
        SELECT use_case, business_process 
        FROM product_view 
        WHERE feature_id IN {affected_features}
      output: "affected_processes"

    - step: 4
      name: "识别影响的数据实体"
      action: "通过技术→数据映射，找到此 API 操作的数据实体"
      query: |
        SELECT entity, operation 
        FROM technical_to_data.api_to_data_entity 
        WHERE api_id = {api_id}
      output: "affected_entities"

    - step: 5
      name: "识别影响的下游数据流"
      action: "检查数据实体关联的数据流"
      query: |
        SELECT data_flow 
        FROM data_flows 
        WHERE source_entity IN {affected_entities}
      output: "affected_data_flows"

    - step: 6
      name: "生成影响报告"
      output_format:
        blast_radius_report:
          api_id: "{api_id}"
          change_type: "{change_type}"
          is_breaking: "{is_breaking}"
          impact_summary:
            services_affected: "{count(affected_services)}"
            features_affected: "{count(affected_features)}"
            processes_affected: "{count(affected_processes)}"
            data_entities_affected: "{count(affected_entities)}"
            data_flows_affected: "{count(affected_data_flows)}"
          risk_level: "auto-calculated"  # low/medium/high/critical
          details:
            affected_services: "{affected_services}"
            affected_features: "{affected_features}"
            affected_processes: "{affected_processes}"
            affected_entities: "{affected_entities}"
            affected_data_flows: "{affected_data_flows}"
          recommended_actions:
            - "..."
```

```yaml
# 05-cross-reference/consistency-rules/consistency-checker.yaml

consistency_checker:
  version: "1.0"
  description: "跨视角一致性检查规则，可集成到 CI/CD 流程中"

  rules:
    # ====== 规则1：每个限界上下文必须有对应的微服务 ======
    - rule_id: "CR-001"
      name: "限界上下文-微服务映射完整性"
      severity: "error"
      check: |
        FOR EACH bc IN business-view/bounded-contexts
        WHERE bc.status = 'active'
        ASSERT EXISTS mapping IN business-to-technical
        WHERE mapping.bounded_context = bc.id
          AND mapping.microservices IS NOT EMPTY
      message: "限界上下文 {bc.id} 没有对应的微服务映射"

    # ====== 规则2：每个微服务必须归属于某个限界上下文 ======
    - rule_id: "CR-002"
      name: "微服务-限界上下文归属完整性"
      severity: "error"
      check: |
        FOR EACH ms IN technical-view/services
        WHERE ms.status = 'active'
        ASSERT EXISTS mapping IN business-to-technical
        WHERE ms.id IN mapping.microservices
      message: "微服务 {ms.id} 没有归属的限界上下文"

    # ====== 规则3：数据实体所有权唯一性 ======
    - rule_id: "CR-003"
      name: "数据实体单一所有权"
      severity: "error"
      check: |
        FOR EACH de IN data-view/entities
        ASSERT COUNT(owners IN business-to-data WHERE owner_type = 'primary') <= 1
      message: "数据实体 {de.id} 有多个主要所有者，违反单一所有权原则"

    # ====== 规则4：领域事件必须有技术实现 ======
    - rule_id: "CR-004"
      name: "领域事件实现完整性"
      severity: "warning"
      check: |
        FOR EACH evt IN business-view/events
        WHERE evt.status = 'active'
        ASSERT EXISTS impl IN technical-view/events
        WHERE impl.domain_event_ref = evt.id
      message: "领域事件 {evt.id} 缺少技术实现映射"

    # ====== 规则5：产品功能必须有 API 支撑 ======
    - rule_id: "CR-005"
      name: "产品功能-API 映射完整性"
      severity: "warning"
      check: |
        FOR EACH ft IN product-view/features
        WHERE ft.status IN ['active', 'developing']
        ASSERT EXISTS mapping IN product-to-technical
        WHERE mapping.feature = ft.id
          AND mapping.apis IS NOT EMPTY
      message: "产品功能 {ft.id} 没有对应的 API 映射"

    # ====== 规则6：联邦索引中的实体引用有效性 ======
    - rule_id: "CR-006"
      name: "实体引用有效性"
      severity: "error"
      check: |
        FOR EACH ref IN all-cross-references
        ASSERT EXISTS entity IN federated-index
        WHERE entity.id = ref.target_id
          AND entity.status != 'deleted'
      message: "引用 {ref.target_id} 指向不存在或已删除的实体"

    # ====== 规则7：ID 全局唯一性 ======
    - rule_id: "CR-007"
      name: "实体 ID 全局唯一"
      severity: "error"
      check: |
        FOR ALL entities IN federated-index
        ASSERT UNIQUE(entity.id)
      message: "发现重复的实体 ID: {duplicate_ids}"

    # ====== 规则8：废弃实体不应被活跃实体引用 ======
    - rule_id: "CR-008"
      name: "废弃实体引用检查"
      severity: "warning"
      check: |
        FOR EACH entity IN federated-index
        WHERE entity.status = 'deprecated'
        ASSERT NOT EXISTS active_ref
        WHERE active_ref.target = entity.id
          AND active_ref.source.status = 'active'
      message: "活跃实体 {active_ref.source} 引用了已废弃的 {entity.id}"
```
